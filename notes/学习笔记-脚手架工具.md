// 实现这个项目的构建任务
const { series, parallel } = require('gulp')
const fs = require('fs')

const task1 = done => {
  setTimeout(() => {
    console.log('tasks1  working...')
    done()
  }, 1000)
}

const task2 = done => {
  setTimeout(() => {
    console.log('tasks2  working...')
    done()
  }, 1000)
}

const task3 = done => {
  setTimeout(() => {
    console.log('tasks3  working...')
    done()
  }, 1000)
}

// 串行
exports.foo = series(task1, task2, task3)
// 并行
exports.bar = parallel(task1, task2, task3)

// 异步
exports.callback = done => {
  console.log('callback task~')
  done()
}

exports.callback_error = done => {
  console.log('callback_error')
  // 一旦报错，后续任务不会再执行
  done(new Error('task failed!'))
}

exports.promise = () => {
  console.log('promise tasks~')
  return Promise.resolve()
}

exports.promise_error = () => {
  console.log('promise tasks~')
  // 同样，后续任务不会再执行
  return Promise.reject(new Error('task failed~'))
}

const timeout = time => {
  return new Promise(resolve => {
    setTimeout(resolve, time)
  })
}

exports.async = async () => {
  await timeout(1000)
  console.log('async task~')
}

Gulp 构建过程核心工作原理

输入（读取文件） -> 加工（压缩文件） -> 输出（写入文件）

```js
// 实现这个项目的构建任务
const fs = require('fs')
const { Transform } = require('stream')

exports.default = () => {
  // 文件读取流
  const read = fs.createReadStream('normalize.css')
  // 文件写入流
  const write = fs.createWriteStream('normalize.min.css')
  // 文件转换流
  const tansform = new Transform({
    transform: (chunk, encoding, callback) => {
      // 核心转换过程
      // chunk => 读取流中读取到的内容(Buffer)
      const input = chunk.toString()
      const output = chunk.replace(/\s+/g, '').replace(/\/\*.+?\*\//g, '')
      // 错误优先，无错误，用 null 代替
      callback(null, output)
    }
  })

  // 把读取出来的文件流导入写入流
  read
    .pipe(transform) // 转换
    .pipe(write) // 写入

  return read
}

```

- 监听变化以及构建优化

```js
// 从 gulp 中解构出 watch 对文件进行监听
// 实现这个项目的构建任务
const { src, dest, parallel, series, watch } = require('gulp')

const clean = () => {
  return del(['dist'])
}

const serve = () => {
  watch('src/assets/styles/*.scss', style)
  watch('src/assets/script/*.js', script)
  watch('src/*.html', page)
  // watch('src/assets/image/**', image)
  // watch('src/assets/fonts/**', font)
  // watch('public/**', extra)
  watch([
    'src/assets/images/**',
    'src/assets/fonts/**',
    'public/**'
  ], bs.reload)

  bs.init({
    notify: false,
    port: 2080,
    // 启动后自动打开浏览器
    open: true,
    // 监听文件改动
    files: 'dist/**',
    // files 可以通过在各个任务末尾添加 pipe(bs.reload({ stream: true })) 替代
    server: {
      baseDir: ['dist', 'src', 'public'],
      // 添加特殊路由，优先于 baseDir
      routes: {
        '/node_modules': 'node_modules'
      }
    }
  })
}

const compile = parallel(style, script, page)

// 上线之前执行的任务
const build = series(clean, parallel(compile, image, font, extra))

const develop = series(compile, serve)

module.exports = {
  clean,
  compile,
  build,
  develop
}

```
